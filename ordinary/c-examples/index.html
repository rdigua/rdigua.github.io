#include <stdio.h>

void main()
{
      void  exchange(int *q1, int *q2, int *q3);

      int a, b, c, *p1, *p2, *p3;

      scanf("%d %d %d", &a, &b, &c);
      p1 = &a;
      p2 = &b; 
      p3 = &c;

      exchange(p1, p2, p3);  //确保a > b > c
      printf("%d %d %d\n", a, b, c);
}


void  exchange(int *q1, int *q2, int *q3) //int *q1 = p1;
{
      void swap(int *pt1, int *pt2); //用于交换&……&%

      if( *q1 < *q2 )
      {
            swap(q1, q2);
      }
      if( *q1 < *q3 )
      {
            swap(q1, q3);
      }
      if( *q2 < *q3 )
      {
            swap(q2, q3);
      }
}
 
void  swap(int *pt1, int *pt2)
{
      int temp;

      temp = *pt1;
      *pt1 = *pt2;
      *pt2 = temp;
}

#include <stdio.h>

int max, min;      /*全局变量*/

void max_min_value(int array[], int n);

void main()
{
      int i, number[10];

      printf("enter 10 integer umbers:\n");

      for( i=0; i < 10; i++)
      {
            scanf("%d", &number[i]);
      }

      max_min_value(number, 10);

      printf("\nmax=%d, min=%d\n", max, min);

}

void max_min_value(int array[], int n)
{
      int *p, *array_end;
      
      array_end = array + n;
      
      max = min = *array;
      
      for( p=array+1; p < array_end; p++)
      {
            if( *p > max )
            {
                  max = *p;
            }
            else if( *p < min ) 
            {
                  min = *p;
            }     
      }
}
